# 1. Bring the interface up

sudo ip addr add 192.168.2.100/24 dev enp1s0

# 3. Confirm
ip addr show enp2s0
# … you should now see inet 192.168.2.100/24 …

# 4. Try pinging your EVB:
ping 192.168.2.123
